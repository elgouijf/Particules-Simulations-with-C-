7def load_frames(filename="frames/frames.txt"):
8 """
9 Charge les données des frames de simulation depuis un fichier texte.
10
11 Le fichier doit contenir cinq colonnes :
12 - identifiant de frame
13 - x
14 - y
15 - z
16 - type de particule
17
18 Parameters
19 ----------
20 filename : str
21 Chemin vers le fichier contenant les frames.
22
23 Returns
24 -------
25 tuple
26 data : np.ndarray
27 Tableau contenant toutes les données.
28 frame_ids : np.ndarray
29 Identifiants uniques des frames.
30
31 Raises
32 ------
33 FileNotFoundError
34 Si le fichier n'existe pas.
35 """
36 if not os.path.exists(filename):
37 raise FileNotFoundError(f"Fichier introuvable : {filename}")
38
39 data = np.loadtxt(filename)
40 frame_ids = np.unique(data[:, 0]).astype(int)
