368void univers::calcule_forces(){
369 // Remise à zéro des forces
370 for (particule* p : particules) {
371 p->setForce(vecteur());
372 }
373
374 const double r_cut2 = r_cut * r_cut;
375 const double sigma2 = sigma * sigma;
376 const double coeff = 24.0 * this->eps;
377
378 // Parcours des cellules
379 for (const cellule& c : this->cellules) {
380 const auto& parts = c.getParticules();
381
382 for (const cellule* v : c.getVoisins()){
383 // Évite de traiter deux fois la même paire de cellules
384 if (v < &c){
385 continue;
386 }
387
388 if (v == &c){
389 // Interactions internes à une même cellule
390 for (size_t i = 0; i < parts.size(); ++i) {
391 for (size_t j = i + 1; j < parts.size(); ++j) {
392
393 particule* pi = parts[i];
394 particule* pj = parts[j];
395
396 const vecteur& posi = pi->getPosition();
397 const vecteur& posj = pj->getPosition();
398
399 double dx = posj.getX() - posi.getX();
400 double dy = posj.getY() - posi.getY();
401 double dz = posj.getZ() - posi.getZ();
402
403 double dist2 = dx * dx + dy * dy + dz * dz + 1e-12;
404
405 if (dist2 > r_cut2) continue;
406
407 double sr2 = sigma2 / dist2;
408 double sr6 = sr2 * sr2 * sr2;
409
410 double coeff_force = coeff * sr6 * (1.0 - 2.0 * sr6) / dist2;
411
412 pi->ajouterForce(dx * coeff_force, dy * coeff_force, dz * coeff_force);
413 pj->ajouterForce(-dx * coeff_force, -dy * coeff_force, -dz * coeff_force);
414 }
415 }
416 }
417 else {
418 // Interactions entre deux cellules voisines distinctes
419 const auto& vois = v->getParticules();
420
421 for (particule* pi : parts) {
422 const vecteur& posi = pi->getPosition();
423
424 for (particule* pj : vois) {
425 const vecteur& posj = pj->getPosition();
426
427 double dx = posj.getX() - posi.getX();
428 double dy = posj.getY() - posi.getY();
429 double dz = posj.getZ() - posi.getZ();
430
431 double dist2 = dx * dx + dy * dy + dz * dz + 1e-12;
432
433 if (dist2 > r_cut2) continue;
434
435 double sr2 = sigma2 / dist2;
436 double sr6 = sr2 * sr2 * sr2;
437
438 double coeff_force = coeff * sr6 * (1.0 - 2.0 * sr6) / dist2;
439
440 pi->ajouterForce(dx * coeff_force, dy * coeff_force, dz * coeff_force);
441 pj->ajouterForce(-dx * coeff_force, -dy * coeff_force, -dz * coeff_force);
442 }
443 }
444 }
445 }
446 }
